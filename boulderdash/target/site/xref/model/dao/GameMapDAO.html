23  
24          final CallableStatement callStatement = prepareCall("select * from map where level = (?)");
25          callStatement.setInt(1, level);
26          if (callStatement.execute()) {
27              final ResultSet result = callStatement.getResultSet();
28  
29      		GameMap gameMap = new GameMap(10,10);
30              for (boolean isResultLeft = result.first(); isResultLeft; isResultLeft = result.next()) {
31  				Position position = new Position(result.getInt(2), result.getInt(3));
32  				Cell cell = null;
33  				if("R".equals(result.getString(4))) {
34  					cell = new Rock(position);
35  				} else if ("P".equals(result.getString(4))) {
36  					cell = new Player(position);
37  				} else if ("M".equals(result.getString(4))) {
38  					cell = new Wall(position);
39  				}
40  				gameMap.getMap().put(position, cell);
41              }
42              result.close();
43              return gameMap;
44          }
45          return null;
